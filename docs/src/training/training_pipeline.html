 32def create_datasets(df: DataFrame, label_encoder: LabelEncoder, tokenizer: AutoTokenizer, encoded: bool, max_length=512, seed=42) -> tuple[ProductIterableDataset, ProductIterableDataset, ProductIterableDataset]:
 33    """
 34    Divides the data into train, validation, and test and creates PyTorch Datasets.
 35
 36    Args:
 37        df (DataFrame): DataFrame containing the preprocessed data.
 38        label_encoder (LabelEncoder): Encoder for the labels.
 39        tokenizer (AutoTokenizer): BERT tokenizer.
 40        encoded (bool): Whether the data is already encoded.
 41        max_length (int): Maximum length of the input sequence for BERT Tokenizer. Defaults to 512.
 42        seed (int): Random seed for reproducibility. Defaults to 42.
 43
 44    Returns:
 45        tuple[ProductIterableDataset, ProductIterableDataset, ProductIterableDataset]: Train, validation, and test datasets.
 46    """
 47
 48    # Create a Spark DataFrame for each data split
 49    train_df, test_df = df.randomSplit([0.8, 0.2], seed=seed)
 50    train_df, val_df = train_df.randomSplit([0.8, 0.2], seed=seed)
 51
 52    # Shuffle the data for better training
 53    train_df = train_df.orderBy(rand(seed=seed))
 54    val_df = val_df.orderBy(rand(seed=seed))
 55    test_df = test_df.orderBy(rand(seed=seed))
 56
 57    train_dataset = ProductIterableDataset(
 58        train_df, label_encoder, tokenizer, encoded, max_length)
 59    val_dataset = ProductIterableDataset(
 60        val_df, label_encoder, tokenizer, encoded, max_length)
 61    test_dataset = ProductIterableDataset(
 62        test_df, label_encoder, tokenizer, encoded, max_length)
 63
 64    return train_dataset, val_dataset, test_dataset

 68    """

158def plot_confusion_matrix(cm: np.ndarray, labels: list[str], log: bool = True):
159    """
160    Plots and logs the confusion matrix.
161
162    Args:
163        cm (np.ndarray): Confusion matrix.
164        labels (list[str]): List of class labels.
165        log (bool): Whether to log the plot to MLflow. Defaults to True.
166    """
167    try:
168        cm_percentage = cm * 100
169        plt.figure(figsize=(20, 12))
170        sns.heatmap(cm_percentage, annot=True, fmt=".1f", cmap='Blues',
171                    annot_kws={"size": 10, "ha": 'center', "va": 'center'})
172        plt.title('Confusion Matrix')
173        plt.ylabel('Actual Labels')
174        plt.xlabel('Predicted Labels')
175        tick_marks = np.arange(len(labels))
176        plt.xticks(tick_marks, labels, rotation=45)
177        # Adding an offset and aligning vertically
178        plt.yticks(tick_marks + 0.5, labels, rotation=0, va='center')
179        plt.tight_layout()
180        plt.savefig('confusion_matrix.png')
181        plt.close()
182        if log:
183            mlflow.log_artifact('confusion_matrix.png')
184    except Exception as e:
185        logger.error("Failed to plot or log confusion matrix: %s",
186                     str(e), exc_info=True)
187        raise

190def train(loader: DataLoader, model: AutoModelForSequenceClassification, optimizer: AdamW, scheduler: LambdaLR, device: str, epoch: int, num_epochs: int):
191    """
192    Trains the model for one epoch.
193
194    Args:
195        loader (DataLoader): PyTorch DataLoader for training data.
196        model (AutoModelForSequenceClassification): BERT model for text classification.
197        optimizer (AdamW): Optimizer for training.
198        scheduler (LambdaLR): Learning rate scheduler.
199        device (str): Device to use for training.
200        epoch (int): Current epoch number.
201        num_epochs (int): Total number of epochs.
202    """
203
204    loader_len = len(loader)
205    model.train()
206    total_loss = 0
207    total_train_acc = 0.0
208    # Dictionary to count occurrences of each label
209    label_counter = defaultdict(int)
210    for i, batch in enumerate(loader):
211        optimizer.zero_grad()
212        input_ids, attention_mask, labels = batch
213        input_ids = input_ids.to(device)
214        attention_mask = attention_mask.to(device)
215        labels = labels.to(device)
216        outputs = model(
217            input_ids, attention_mask=attention_mask, labels=labels)
218        loss = outputs.loss
219        total_loss += loss.item()
220        loss.backward()
221        optimizer.step()
222        scheduler.step()
223        logits = outputs.logits
224        logits = logits.detach().cpu().numpy()
225        label_ids = labels.to('cpu').numpy()
226        batch_acc = accuracy_score(label_ids, logits.argmax(axis=1))
227        total_train_acc += batch_acc
228        # Update label counts
229        unique, counts = np.unique(label_ids, return_counts=True)
230        for label, count in zip(unique, counts):
231            label_counter[label] += count
232
233        if (i+1) % 100 == 0 or (i+1) == loader_len:
234            step_loss = total_loss / i
235            step_acc = total_train_acc / i
236            logger.info(
237                'Epoch %d/%d, Step %d/%d, Loss: %.4f, Accuracy: %.2f%%',
238                epoch+1, num_epochs, i, loader_len, step_loss, step_acc * 100)
239            mlflow.log_metrics(
240                {'Step Loss': step_loss, 'Step Accuracy': step_acc}, step=epoch * loader_len + i)
241    logger.info("--------------------")
242    logger.info("Epoch %d/%d, Average Training Loss: %.6f, Average Training Accuracy: %.2f%%",
243                epoch + 1, num_epochs, total_loss / loader_len, total_train_acc / loader_len * 100)
244    logger.info("--------------------")
245    mlflow.log_metrics({'Training Loss': total_loss / loader_len,
246                       'Training Accuracy': total_train_acc / loader_len}, step=epoch+1)

249def evaluate(loader: DataLoader, model: AutoModelForSequenceClassification, device: str, epoch: int, num_epochs: int):
250    """
251    Evaluates the model on the validation set after each training epoch.
252
253    Args:
254        loader (DataLoader): PyTorch DataLoader for validation data.
255        model (AutoModelForSequenceClassification): BERT model for text classification.
256        device (str): Device to use for evaluation.
257        epoch (int): Current epoch number.
258        num_epochs (int): Total number of epochs.
259
260    Returns:
261        float: Validation loss.
262    """
263    loader_len = len(loader)
264    model.eval()
265    eval_loss, eval_acc = 0, 0
266    try:
267        with torch.no_grad():
268            for batch in tqdm(loader, desc="Validation", unit="batch"):
269                input_ids, attention_mask, labels = batch
270                input_ids = input_ids.to(device)
271                attention_mask = attention_mask.to(device)
272                labels = labels.to(device)
273                outputs = model(
274                    input_ids, attention_mask=attention_mask, labels=labels)
275                loss = outputs.loss
276                eval_loss += loss.item()
277                logits = outputs.logits
278                logits = logits.detach().cpu().numpy()
279                label_ids = labels.to('cpu').numpy()
280                eval_acc += accuracy_score(label_ids, logits.argmax(axis=1))
281
282        eval_loss = eval_loss / loader_len
283        eval_acc = eval_acc / loader_len
284        logger.info("--------------------")
285        logger.info(
286            'Epoch %d/%d, Validation Loss: %.4f, Validation Accuracy: %.2f%%',
287            epoch+1, num_epochs, eval_loss, eval_acc * 100)
288        logger.info("--------------------")
289        mlflow.log_metrics({'Validation Loss': eval_loss,
290                            'Validation Accuracy': eval_acc}, step=epoch+1)
291        return eval_loss, eval_acc
292    except Exception as e:
293        logger.error("Error during model evaluation: %s", e, exc_info=True)
294        raise

297def test(loader: DataLoader, decoded_categories: list[str], model: AutoModelForSequenceClassification, device: str):
298    """
299    Tests the model on the test set.
300
301    Args:
302        loader (DataLoader): PyTorch DataLoader for test data.
303        decoded_categories (list[str]): List of decoded category labels.
304        model (AutoModelForSequenceClassification): BERT model for text classification.
305        device (str): Device to use for testing.
306
307    Returns:
308        dict: Dictionary containing the test metrics.
309    """
310    model.eval()
311    all_labels = []
312    all_predictions = []
313    try:
314        with torch.no_grad():
315            for batch in tqdm(loader, desc="Testing", unit="batch"):
316                input_ids, attention_mask, labels = batch
317                input_ids = input_ids.to(device)
318                attention_mask = attention_mask.to(device)
319                labels = labels.to(device)
320                outputs = model(
321                    input_ids, attention_mask=attention_mask, labels=labels)
322                logits = outputs.logits
323                logits = logits.detach().cpu().numpy()
324                label_ids = labels.to('cpu').numpy()
325                all_labels.extend(label_ids.tolist())
326                all_predictions.extend(logits.argmax(axis=1).tolist())
327
328        acc = accuracy_score(all_labels, all_predictions)
329        f1 = f1_score(all_labels, all_predictions, average='weighted')
330        precision = precision_score(
331            all_labels, all_predictions, average='weighted', zero_division=0)
332        recall = recall_score(all_labels, all_predictions,
333                              average='weighted', zero_division=0)
334        cm = confusion_matrix(all_labels, all_predictions)
335        cm_sum = cm.sum(axis=1)[:, np.newaxis]
336        cm_normalized = np.divide(
337            cm.astype('float'), cm_sum, where=cm_sum != 0)
338        cm_normalized = np.nan_to_num(cm_normalized)
339        # cm_normalized = cm.astype('float') / cm.sum(axis=1)[:, np.newaxis]
340
341        logger.info('Test Accuracy: %.2f%%, F1 Score: %.2f, Precision: %.2f, Recall: %.2f',
342                    acc * 100, f1, precision, recall)
343        test_metrics = {
344            'Accuracy': acc,
345            'F1 Score': f1,
346            'Precision': precision,
347            'Recall': recall,
348        }
349        # Log the test metrics
350        for key, value in test_metrics.items():
351            mlflow.log_metric(f"Test {key}", value)
352        # Plot and log the confusion matrix
353        plot_confusion_matrix(cm_normalized, decoded_categories)
354
355        # Log the best model
356        mlflow.pytorch.log_model(model, "model")
357        return test_metrics
358    except Exception as e:
359        logger.error("Error during model testing: %s", e, exc_info=True)
360        raise

363def run(config='src/training/configs/default_config.yml', **overrides):
364    """
365    Function to run the training pipeline.
366
367    Args:
368        config (str): Path to the configuration file. Defaults to 'src/training/default_config.yml'.
369            input_data_path (str): Path to the input JSONL.gz file.
370            input_data_encoded (bool): Whether the data is already encoded. Defaults to False.
371            bert_model_name (str): Name of the BERT model to use for tokenization. Defaults to 'bert-base-uncased'.
372            data_load ('memory' | 'distributed'): Method to load the data. Defaults to 'distributed'.
373            trainable_layers (int | None): Number of layers to keep trainable. None for all layers. Defaults to None.
374            num_epochs (int): Number of epochs to train the model. Defaults to 3.
375            batch_size (int): Batch size for training and validation. Defaults to 8.
376            test_batch_size (int): Batch size for testing. Defaults to 32.
377            learning_rate (float): Learning rate for the optimizer. Defaults to 1e-5.
378            optimizer (str): Optimizer to use for training. Defaults to 'AdamW'.
379            seed (int): Random seed for reproducibility. Defaults to 42.
380            device (str): Device to use for training. Defaults to 'cuda'.
381            data_fraction (float): Fraction of the data to sample for training. Defaults to 1.0.
382    """
383    try:
384        logger.info("Logging in to MLflow server hosted at Databricks CE.")
385        mlflow.login()
386        mlflow.set_tracking_uri(SETTINGS["MLFLOW_TRACKING_URI"])
387
388        logger.info("Setting up MLflow experiment.")
389        mlflow.set_experiment(SETTINGS["MLFLOW_EXPERIMENT_NAME"])
390
391        config = utils.load_config(config)
392        config.update(overrides)
393
394        input_data_path = config.get(
395            "input_data_path", "data/preprocessed/amz_products_small_preprocessed_v1.parquet")
396        bert_model_name = config.get("bert_model_name", "bert-base-uncased")
397        data_load = config.get("data_load", "distributed")
398        input_data_encoded = config.get("input_data_encoded", False)
399        trainable_layers = config.get("trainable_layers", None)
400        num_epochs = config.get("num_epochs", 3)
401        batch_size = config.get("batch_size", 8)
402        test_batch_size = config.get("test_batch_size", 32)
403        learning_rate = float(config.get("learning_rate", 1e-5))
404        optimizer = config.get("optimizer", "AdamW")
405        seed = config.get("seed", 42)
406        device = config.get("device", torch.device(
407            'cuda' if torch.cuda.is_available() else 'cpu'))
408        data_fraction = config.get("data_fraction", 1.0)
409
410        if device == 'cuda':
411            logger.info("Checking if CUDA is available.")
412            device = torch.device(
413                'cuda' if torch.cuda.is_available() else 'cpu')
414            # device = torch.device('cpu')
415            logger.info("Using device: %s.", device)
416
417        logger.info("Setting random seed: %d.", seed)
418        utils.set_seed(seed, device=device)
419        current_time = datetime.datetime.now().strftime("%Y-%m-%d %H:%M")
420
421        with mlflow.start_run(run_name=f"Run {current_time}"):
422            params = {
423                "input_data_path": input_data_path,
424                "bert_model_name": bert_model_name,
425                "data_load": data_load,
426                "input_data_encoded": str(input_data_encoded),
427                "trainable_layers": trainable_layers,
428                "num_epochs": num_epochs,
429                "batch_size": batch_size,
430                "test_batch_size": test_batch_size,
431                "learning_rate": learning_rate,
432                "optimizer": optimizer,
433                "seed": seed,
434                "device": str(device),
435                "data_fraction": data_fraction
436            }
437            mlflow.log_params(params)
438            logger.info("Starting run with configuration:")
439            logger.info("\n%s", pformat(params))
440
441            # Encode categories
442            label_encoder = LabelEncoder()
443            label_encoder.fit(CATEGORIES)
444
445            # Create BERT tokenizers
446            logger.info("Creating BERT tokenizer for model: %s.",
447                        bert_model_name)
448            tokenizer = AutoTokenizer.from_pretrained(bert_model_name)
449
450            logger.info("Creating the BERT model for text classification.")
451            # Create the Bert model for text classification
452            model = AutoModelForSequenceClassification.from_pretrained(
453                bert_model_name, num_labels=len(CATEGORIES))
454
455            model.to(device)
456
457            # Freeze all layers except the last trainable_layers
458            if trainable_layers is not None and trainable_layers != 'None':
459                trainable_layers = int(trainable_layers)
460                logger.info(
461                    "Freezing all layers except the last %d layers.", trainable_layers)
462                total_params, trainable_params = utils.freeze_layers(
463                    model, trainable_layers)
464                logger.info("Total model parameters: %d.", total_params)
465                logger.info("Total trainable parameters: %d.",
466                            trainable_params)
467            # Create the optimizer
468            optimizer = AdamW(model.parameters(), lr=learning_rate,  eps=1e-8)
469
470            # Load the data
471            if data_load == 'distributed':
472                logger.info(
473                    "Extracting preprocessed data from file to a distributed PySpark DataFrame.")
474
475                df = utils.read_parquet_to_pyspark(input_data_path)
476                data_size = df.count()
477                logger.info("Data extracted with size: %d rows", data_size)
478
479                if data_fraction < 1.0:
480                    logger.info("Sampling data with fraction: %f.",
481                                data_fraction)
482                    df = df.sample(fraction=data_fraction, seed=seed)
483                    data_size = df.count()
484                    logger.info("Data sampled with size: %d rows", data_size)
485
486                logger.info("Creating training and validation datasets.")
487                train_dataset, val_dataset, test_dataset = create_datasets(
488                    df, label_encoder, tokenizer, input_data_encoded, 512, seed=seed)
489
490                logger.info(
491                    "Creating the train and validation dataloaders with batch size: %d.", batch_size)
492                logger.info(
493                    "Creating the test dataloader with batch size: %d.", test_batch_size)
494                # Create the dataloaders
495                train_loader, val_loader, test_loader = create_loaders(
496                    train_dataset, val_dataset, test_dataset, batch_size, test_batch_size)
497            else:
498                logger.info(
499                    "Extracting preprocessed data from file to an in memory Pandas DataFrame.")
500                df = utils.read_parquet_to_pandas(input_data_path)
501                data_size = len(df)
502                logger.info("Data extracted with size: %d rows", data_size)
503
504                if data_fraction < 1.0:
505                    logger.info("Sampling data with fraction: %f.",
506                                data_fraction)
507                    df = df.sample(frac=data_fraction, random_state=seed)
508                    data_size = len(df)
509                    logger.info("Data sampled with size: %d rows", data_size)
510
511                logger.info("Creating training and validation datasets.")
512                train_dataset, val_dataset, test_dataset = create_in_memory_datasets(
513                    df, label_encoder, tokenizer, input_data_encoded, 512, seed=seed)
514
515                logger.info(
516                    "Creating the train and validation dataloaders with batch size: %d.", batch_size)
517                logger.info(
518                    "Creating the test dataloader with batch size: %d.", test_batch_size)
519                # Create the dataloaders
520                train_loader, val_loader, test_loader = create_in_memory_loaders(
521                    train_dataset, val_dataset, test_dataset, batch_size, test_batch_size)
522
523            # Create the learning rate scheduler
524            total_steps = len(train_loader) * num_epochs
525            # Set the warmup steps to 10% of the total steps
526            warmup_steps = int(0.1 * total_steps)
527            scheduler = get_linear_schedule_with_warmup(
528                optimizer, num_warmup_steps=warmup_steps, num_training_steps=total_steps)
529
530            logger.info("Training the model.")
531
532            best_model_state = model.state_dict()
533            best_eval_loss = float('inf')
534            try:
535                # Training loop
536                for epoch in range(num_epochs):
537                    # Train one epoch
538                    # Shuffle the data after the first epoch if using IterableDataset
539                    if (data_load == 'distributed' and epoch > 0):
540                        train_loader.dataset.reshuffle(seed=seed)
541                        train_loader = DataLoader(
542                            train_loader.dataset, batch_size=batch_size)
543
544                    train(train_loader, model, optimizer, scheduler,
545                          device, epoch, num_epochs)
546                    # Validation after each epoch
547                    eval_loss, eval_acc = evaluate(
548                        val_loader, model, device, epoch, num_epochs)
549
550                    # Save the model if it has the best validation loss so far
551                    if eval_loss < best_eval_loss:
552                        best_eval_loss = eval_loss
553                        best_model_state = model.state_dict()
554                        logger.info(
555                            "New best model saved in epoch %d with validation loss: %.4f and accuracy: %.2f%%",
556                            epoch+1, eval_loss, eval_acc * 100)
557

467                            trainable_params)
468            # Create the optimizer
469            optimizer = AdamW(model.parameters(), lr=learning_rate,  eps=1e-8)
470
471            # Load the data
472            if data_load == 'distributed':
473                logger.info(
474                    "Extracting preprocessed data from file to a distributed PySpark DataFrame.")
475
476                df = utils.read_parquet_to_pyspark(input_data_path)
477                data_size = df.count()
478                logger.info("Data extracted with size: %d rows", data_size)
479
480                if data_fraction < 1.0:
481                    logger.info("Sampling data with fraction: %f.",
482                                data_fraction)
483                    df = df.sample(fraction=data_fraction, seed=seed)
484                    data_size = df.count()
485                    logger.info("Data sampled with size: %d rows", data_size)
486
487                logger.info("Creating training and validation datasets.")
488                train_dataset, val_dataset, test_dataset = create_datasets(
489                    df, label_encoder, tokenizer, input_data_encoded, 512, seed=seed)
490
491                logger.info(
492                    "Creating the train and validation dataloaders with batch size: %d.", batch_size)
493                logger.info(
494                    "Creating the test dataloader with batch size: %d.", test_batch_size)
495                # Create the dataloaders
496                train_loader, val_loader, test_loader = create_loaders(
497                    train_dataset, val_dataset, test_dataset, batch_size, test_batch_size)
498            else:
499                logger.info(
500                    "Extracting preprocessed data from file to an in memory Pandas DataFrame.")
501                df = utils.read_parquet_to_pandas(input_data_path)
502                data_size = len(df)
503                logger.info("Data extracted with size: %d rows", data_size)
504
505                if data_fraction < 1.0:
506                    logger.info("Sampling data with fraction: %f.",
507                                data_fraction)
508                    df = df.sample(frac=data_fraction, random_state=seed)
509                    data_size = len(df)
510                    logger.info("Data sampled with size: %d rows", data_size)
511
512                logger.info("Creating training and validation datasets.")
513                train_dataset, val_dataset, test_dataset = create_in_memory_datasets(
514                    df, label_encoder, tokenizer, input_data_encoded, 512, seed=seed)
515
516                logger.info(
517                    "Creating the train and validation dataloaders with batch size: %d.", batch_size)
518                logger.info(
519                    "Creating the test dataloader with batch size: %d.", test_batch_size)
520                # Create the dataloaders
521                train_loader, val_loader, test_loader = create_in_memory_loaders(
522                    train_dataset, val_dataset, test_dataset, batch_size, test_batch_size)
523
524            # Create the learning rate scheduler
525            total_steps = len(train_loader) * num_epochs
526            # Set the warmup steps to 10% of the total steps
527            warmup_steps = int(0.1 * total_steps)
528            scheduler = get_linear_schedule_with_warmup(
529                optimizer, num_warmup_steps=warmup_steps, num_training_steps=total_steps)
530
531            logger.info("Training the model.")
532
533            best_model_state = model.state_dict()
534            best_eval_loss = float('inf')
535            try:
536                # Training loop
537                for epoch in range(num_epochs):
538                    # Train one epoch
539                    # Shuffle the data after the first epoch if using IterableDataset
540                    if (data_load == 'distributed' and epoch > 0):
541                        train_loader.dataset.reshuffle(seed=seed)
542                        train_loader = DataLoader(
543                            train_loader.dataset, batch_size=batch_size)
544
545                    train(train_loader, model, optimizer, scheduler,
546                          device, epoch, num_epochs)
547                    # Validation after each epoch
548                    eval_loss, eval_acc = evaluate(
549                        val_loader, model, device, epoch, num_epochs)
550
551                    # Save the model if it has the best validation loss so far
552                    if eval_loss < best_eval_loss:
553                        best_eval_loss = eval_loss
554                        best_model_state = model.state_dict()
555                        logger.info(
556                            "New best model saved in epoch %d with validation loss: %.4f and accuracy: %.2f%%",
557                            epoch+1, eval_loss, eval_acc * 100)
558
559                # Load the best model before testing
560                model.load_state_dict(best_model_state)
561                logger.info("Best model loaded for testing.")
562                # Evaluate the best model on test data
563                encoded_categories = label_encoder.transform(CATEGORIES)
564                decoded_categories = label_encoder.inverse_transform(
565                    encoded_categories)
566                test(test_loader, decoded_categories, model, device)
567            except Exception as e:
568                logger.error("Error during model training: %s",
569                             e, exc_info=True)
570                raise
571
572    except Exception as e:
573        logger.error(
574            "An error occurred during the training pipeline: %s", str(e), exc_info=True)
575        raise
